```javascript

// Flows 입력값
// User ID: {{context.user.user_id}}
// Body
{
  "user_metadata": {
    "company_name": "{{fields.company_name}}"
  }
}

// Action 코드
// 제목: Render Progressive Profile Form
exports.onExecutePostLogin = async (event, api) => {
  const FORM_ID = '<FORM_ID 입력>';

  if (
    event.stats.logins_count > 2 && !event.user.user_metadata.company_name
  ) {
    api.prompt.render(FORM_ID);
  }
}
