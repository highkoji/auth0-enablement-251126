exports.onExecutePostLogin = async (event, api) => {
  const FORM_ID = '<본인 FORM_ID>';

  // 조건: 로그인 3회차 이상이면서 회사명이 없는 경우
  if (event.stats.logins_count > 2 && !event.user.user_metadata.company_name) {
    // [수정] return을 붙여서 확실히 폼을 띄우도록 함
    return api.prompt.render(FORM_ID);
  }
};
